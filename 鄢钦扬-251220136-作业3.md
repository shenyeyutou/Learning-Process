"""cpp

#pragma once

#include<iostream>
#include<vector>
using namespace std;

class Pet
{
protected:
	int pet_weight;
	int pet_age;
	static vector<Pet*>allpet;

public:
	Pet(int m, int a) :pet_weight(m), pet_age(a)
	{
		allpet.push_back(this);
	}

	virtual ~Pet()
	{
		for (auto petself = allpet.begin(); petself != allpet.end(); petself++)
		{
			if (*petself == this)
			{
				allpet.erase(petself);
				break;
			}
		}
	}

	int weight() const
	{
		return pet_weight;
	}

	void changeWeight(int w)
	{
		pet_weight += w;
	}

	int age() const
	{
		return pet_age;
	}

	static void growAll()
	{
		for (Pet* p : allpet)
		{
			p->pet_age += 1;
		}
	}

	virtual void meow() const = 0;
};

vector<Pet*>Pet::allpet;

class Cat :public Pet
{
public:
	Cat(int c_m,int c_a):Pet(c_m,c_a){}
	void meow() const override
	{
		cout << "Meow~" << endl;
	}
};

class Dog :public Pet
{
public:
	Dog(int d_m,int d_a):Pet(d_m,d_a){}
	void meow() const override
	{
		cout << "Woof!" << endl;
	}
};
