'''cpp
//头文件

class Matrix
{
private:
	int row;
	int col;
	int** num;
	static int matrix_count;

public:
	Matrix(int r, int c);
	Matrix(const Matrix& other);
	~Matrix();
	int getRows() const;
	int getCols() const;
	int getElement(int i, int j) const;
	void setElement(int i, int j, int val);
	Matrix add(const Matrix& other) const;
	Matrix sub(const Matrix& other) const;
	static int getMatrix_count();
};

#endif

//源文件
#include"Matrix.h"
#include<iostream>
using namespace std;

int Matrix::matrix_count = 0;

Matrix::Matrix(int r, int c) :row(r), col(c)
{
	num = new int* [row];
	for (int i = 0; i < row; i++)
	{
		num[i] = new int[col];
		for (int j = 0; j < col; j++)
		{
			num[i][j] = 0;
		}
	}
	matrix_count++;
}

Matrix::Matrix(const Matrix& other) :row(other.row), col(other.col)
{
	num = new int* [row];
	for (int i = 0; i < row; i++)
	{
		num[i] = new int[col];
		for (int j = 0; j < col; j++)
		{
			num[i][j] = other.num[i][j];
		}
	}
	matrix_count++;
}

Matrix::~Matrix()
{
	for (int i = 0; i < row; i++)
	{
		delete[] num[i];
	}
	delete[] num;
	matrix_count--;
}

int Matrix:: getRows() const
{
	return row;
}

int Matrix::getCols() const
{
	return col;
}

int Matrix::getElement(int i, int j) const
{
	return num[i][j];
}

void Matrix::setElement(int i, int j, int val) 
{
	num[i][j] = val;
}

Matrix Matrix::add(const Matrix& other) const
{
	Matrix result(row, col);
	for (int i = 0; i < row; i++)
	{
		for (int j = 0; j < col; j++)
		{
			result.num[i][j] = num[i][j] + other.num[i][j];
		}
	}
	return result;
}

Matrix Matrix::sub(const Matrix& other) const
{
	Matrix result(row, col);
	for (int i = 0; i < row; i++)
	{
		for (int j = 0; j < col; j++)
		{
			result.num[i][j] = num[i][j] - other.num[i][j];
		}
	}
	return result;
}

int Matrix::getMatrix_count()
{
	return matrix_count;
}
