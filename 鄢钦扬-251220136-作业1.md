'''cpp
#ifndef TABLE_H
#define TABLE_H
#include<iostream>
using namespace std;

class table
{
public:
	virtual void push(int x) = 0;
	virtual void pop() = 0;
	virtual void my_print() = 0;
	virtual  ~table(){}

	struct Node
	{
		int value;
		Node* next;
		Node(int val) :value(val), next(nullptr) {}
	};
	
};

class stack :public table
{
private:
	Node* top;

public:
	stack():top(nullptr){}
	~stack()
	{
		while (top != nullptr)
		{
			Node* a_node = top;
			top = top->next;
			delete a_node;
		}
	}

	void push(int val) override
	{
		Node* new_node = new Node(val);
		new_node->next = top;
		top = new_node;
	}

	void pop() override
	{
		if (top == nullptr)
		{
			cout << "error" << endl;
			return;
		}
		cout << top->value << endl;
		Node* a_node = top;
		top = top->next;
		delete a_node;
		return;
	}

	void my_print() override
	{
		if (top == nullptr)
		{
			cout << "error" << endl;
			return;
		}
		cout << top->value << endl;
		return;
	}
};
